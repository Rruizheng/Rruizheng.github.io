# 防抖和节流

# 概念

  • 防抖：连续的操作，只执行最后一次
  • 节流：在每段时间内，只执行一次，用于降低执行频率

# 场景

  • 防抖
    • 搜索
    • 输入校验
    • 窗口大小调整
  • 节流
    • 滚动加载
    • 联想搜索
    • 表单重复提交？？这个不一定吧

// 优化，使用闭包
function debounce(fn, delay) {
    var timer;
    return function() {
        var _this = this;
        var args = arguments;
        timer && clearTimeout(timer);
        timer = setTimeout(function() {
            // 用apply指向调用debounce的对象，相当于_this.fn(args);
            fn.apply(_this, args);
        }, delay);
    };
}
